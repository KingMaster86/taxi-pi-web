function makeReservation($data, $mysqli) {
    $stmt = $mysqli->prepare("INSERT INTO reservation (date, time, passenger_id, driver_id, 
    pickup_location, drop_location, status, operator_id) VALUES (?, ?, ?, ?, ?, ?, ?, ?)");
    
    // Bind parameters
    $stmt->bind_param("ssiiisii", $data['date'], $data['time'], $data['passenger_id'], $data['driver_id'], $data['pickup_location'], $data['drop_location'], $data['status'], $data['operator_id']);

    // Execute the statement
    $stmt->execute();

    // Close the statement
    $stmt->close();
}
